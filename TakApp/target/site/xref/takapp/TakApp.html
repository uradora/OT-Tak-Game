9   import javafx.application.Application;
10  import javafx.stage.Stage;
11  import javafx.scene.Parent;
12  import javafx.scene.layout.Pane;
13  import javafx.scene.Scene;
14  import javafx.scene.Group;
15  import domain.GameLogic;
16  
17  /**
18   *
19   * @author meriraja
20   */
21  public class TakApp extends Application {
22      
23      //todo: add options for board size
24      public static final int TILE_SIZE = 100;
25      public static final int WIDTH = 4;
26      public static final int HEIGHT = 4;
27      public static Tile[][] gameBoard = new Tile[WIDTH][HEIGHT];
28      
29      public static Group tileGroup = new Group();
30      public static Group pieceGroup = new Group();
31      
32      private GameLogic logic = new GameLogic();
33      
34      private Parent createContent() {
35          Pane root = new Pane();
36          root.setPrefSize(WIDTH * TILE_SIZE, HEIGHT * TILE_SIZE);
37          root.getChildren().addAll(tileGroup, pieceGroup);
38          
39          for (int y = 0; y < HEIGHT; y++) {
40              for (int x = 0; x < WIDTH; x++) {
41                  Tile tile = new Tile(logic, x, y);
42                  gameBoard[x][y] = tile;
43                  
44                  tileGroup.getChildren().add(tile);
45              }
46          }
47          
48          return root;
49      }
50      
51      @Override
52      public void start(Stage primaryStage) throws Exception {
53          Scene scene = new Scene(createContent());
54          primaryStage.setTitle("Tak App");
55          primaryStage.setScene(scene);
56          primaryStage.show();
57      }
58      
59      //todo: what happens here when removing pieces?
60      public static void updateBoard(Tile tile, Piece piece, int x, int y) {
61          gameBoard[x][y] = tile;
62          if (!pieceGroup.getChildren().contains(piece)) {
63              pieceGroup.getChildren().add(piece);
64          }
65      }
66  
67      public static void main(String[] args) { 
68          launch(TakApp.class); }
69   
70  }
